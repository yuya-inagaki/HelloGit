・for文

    処理

    print(i)

・__init__ メソッド
__init__メソッドは、インスタンスを生成した直後に処理を実行することができる。

class MenuItem:
    def __init__(self):
        print('インスタンスが生成されました！')

menu_item1 = MenuItem()
